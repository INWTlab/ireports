\newcommand\BackgroundPic{%
	\put(0,0){%
		\parbox[b][\paperheight]{\paperwidth}{%
			\vfill
			\centering
			\includegraphics[width=\paperwidth,height=\paperheight]{\cover}%
			\vfill
}}}

	\end{tabular}
	
	
	
	\clearpage
